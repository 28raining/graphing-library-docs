## Plot One Line

<pre class="mcode">
x = 1:100;
y = x.^2;
semilogy(x,y)
grid on

fig2plotly()
</pre>

## Specify Tick Locations, Tick Labels, and Axis Labels

<pre class="mcode">
P = 1000; 
npayments = 240;  
rate = 0.08/12; 
mpayment = P*(rate*(1+rate)^npayments)/(((1+rate)^npayments) - 1);
x = 1:240;
y = x * mpayment;
semilogy(x,y);
grid on

fig2plotly()
</pre>

x = 1:50;
y = x.^2;
y(20) = NaN;
semilogy(x,y)
