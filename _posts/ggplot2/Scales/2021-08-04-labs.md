Modify axis, legend, and plot labels and then convert them with ggplotly








<pre class="mcode">
p <- ggplot(mtcars, aes(mpg, wt, colour = cyl)) + geom_point()
p <-  p + labs(colour = "Cylinders")
</pre>


<pre class="mcode">
plotly::ggplotly(p)
</pre>
