Construct sophisticated graphics utilizing <var>+</var> and then convert them with ggplotly

<pre class="mcode">
base <-
 ggplot(mpg, aes(displ, hwy)) +
 geom_point()
p <-  base + geom_smooth()
</pre>


<pre class="mcode">
plotly::ggplotly(p)
</pre>

<pre class="mcode">
## `geom_smooth()` using method = 'loess' and formula 'y ~ x'

base <-

 geom_point()
