# Default jitter

Jittering is useful when you have a discrete position, and a relatively small number of points take up as much space as a boxplot or a bar.

p <-    
 ggplot(mpg, aes(class, hwy)) +
  geom_boxplot(colour = "grey50") +
  geom_jitter()
  
plotly::ggplotly(p)
